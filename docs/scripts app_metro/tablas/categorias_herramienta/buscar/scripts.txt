Modulo: categorias_herramienta
Operacion: buscar

Sentencias SQL:

- Obtener categoria por id
SELECT
    id_ai_categoria_herramienta,
    nombre_categoria,
    descripcion,
    std_reg
FROM categoria_herramienta
WHERE id_ai_categoria_herramienta = :id
LIMIT 1;

- Verificar herramientas activas asociadas
SELECT COUNT(1)
FROM herramienta
WHERE id_ai_categoria_herramienta = :id
  AND std_reg = 1;
