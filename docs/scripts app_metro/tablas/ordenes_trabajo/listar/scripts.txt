Modulo: ordenes_trabajo
Operacion: listar

Sentencias SQL:

-Ordenes de trabajo:
SELECT ot.id_ai_ot, ot.n_ot, ot.id_ai_area, ot.id_user, ot.id_ai_sitio, ot.nombre_trab, ot.fecha, ot.semana, ot.mes, ot.std_reg, det_ord.id_ai_estado, estado.nombre_estado, estado.color FROM orden_trabajo ot LEFT JOIN (SELECT n_ot, MAX(id_ai_estado) AS id_ai_estado FROM detalle_orden GROUP BY n_ot) det_ord ON ot.n_ot = det_ord.n_ot LEFT JOIN estado_ot estado ON det_ord.id_ai_estado = estado.id_ai_estado WHERE ot.std_reg = 1 ORDER BY ot.n_ot ASC;

-Detalles de ordenes de trabajo:
SELECT d.id_ai_detalle, d.n_ot, d.fecha, d.descripcion, d.id_ai_turno, d.id_miembro_cco, d.id_user_act, d.id_miembro_ccf, d.id_ai_estado, d.cant_tec, d.hora_ini_pre, d.hora_fin_pre, d.hora_ini_tra, d.hora_fin_tra, d.hora_ini_eje, d.hora_fin_eje, d.observacion, e.nombre_estado, e.color FROM detalle_orden d LEFT JOIN estado_ot e ON d.id_ai_estado = e.id_ai_estado WHERE d.n_ot = :n_ot ORDER BY d.id_ai_detalle DESC;
