- OT resumen
SELECT
    ot.n_ot,
    ot.fecha,
    ot.semana,
    ot.mes,
    ot.nombre_trab,
    a.nombre_area,
    s.nombre_sitio,
    (
        SELECT e.nombre_estado
        FROM detalle_orden d
        LEFT JOIN estado_ot e
            ON e.id_ai_estado = d.id_ai_estado
        WHERE d.n_ot = ot.n_ot
        ORDER BY d.id_ai_detalle DESC
        LIMIT 1
    ) AS estado_actual,
    (
        SELECT COUNT(1)
        FROM detalle_orden d2
        WHERE d2.n_ot = ot.n_ot
    ) AS total_detalles,
    (
        SELECT COUNT(1)
        FROM herramientaot h
        WHERE h.n_ot = ot.n_ot
    ) AS total_herr
FROM orden_trabajo ot
LEFT JOIN area_trabajo a
    ON a.id_ai_area = ot.id_ai_area
LEFT JOIN sitio_trabajo s
    ON s.id_ai_sitio = ot.id_ai_sitio
WHERE ot.std_reg = 1
ORDER BY ot.fecha DESC, ot.n_ot DESC
LIMIT 500;

Filtros condicionales aplicados sobre la consulta OT resumen / OT detallado

AND ot.n_ot = :n_ot;
AND ot.fecha >= :desde;
AND ot.fecha <= :hasta;
AND ot.id_ai_area = :area;
AND ot.id_ai_sitio = :sitio;
AND EXISTS (
    SELECT 1
    FROM detalle_orden d
    WHERE d.n_ot = ot.n_ot
      AND d.id_ai_estado = :estado
);
AND EXISTS (
    SELECT 1
    FROM detalle_orden d2
    WHERE d2.n_ot = ot.n_ot
      AND d2.id_user_act = :usuario
);

- Detalles de una OT
SELECT
    d.id_ai_detalle,
    d.n_ot,
    d.fecha,
    d.descripcion,
    d.id_ai_turno,
    d.id_miembro_cco,
    d.id_user_act,
    d.id_miembro_ccf,
    d.id_ai_estado,
    d.cant_tec,
    d.hora_ini_pre,
    d.hora_fin_pre,
    d.hora_ini_tra,
    d.hora_fin_tra,
    d.hora_ini_eje,
    d.hora_fin_eje,
    d.observacion,
    t.nombre_turno,
    e.nombre_estado,
    u.user AS tecnico_nombre,
    cco.nombre_miembro AS cco_nombre,
    ccf.nombre_miembro AS ccf_nombre
FROM detalle_orden d
LEFT JOIN turno_trabajo t
    ON t.id_ai_turno = d.id_ai_turno
LEFT JOIN estado_ot e
    ON e.id_ai_estado = d.id_ai_estado
LEFT JOIN user_system u
    ON u.id_user = d.id_user_act
LEFT JOIN miembro cco
    ON cco.id_miembro = d.id_miembro_cco
LEFT JOIN miembro ccf
    ON ccf.id_miembro = d.id_miembro_ccf
WHERE d.n_ot = :id
ORDER BY d.id_ai_detalle ASC;
